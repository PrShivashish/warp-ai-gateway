# Node.js Example

A simple Node.js script using the Warp SDK.

## Setup

```bash
mkdir warp-example && cd warp-example
npm init -y
npm install warp-sdk
```

## `index.ts`

```ts
import { Warp } from "warp-sdk";

const client = new Warp({
  apiKey: process.env.WARP_API_KEY!,
  baseUrl: "http://localhost:4000",
  defaultModel: "google/gemini-2.0-flash",
});

async function main() {
  // List available models
  console.log("📋 Available models:");
  const models = await client.models();
  models.forEach((m) => console.log(`  - ${m.slug}`));
  console.log();

  // Non-streaming chat
  console.log("💬 Chat (non-streaming):");
  const response = await client.chat({
    messages: [{ role: "user", content: "What is Node.js in one sentence?" }],
  });
  console.log(`  ${response.completions.choices[0].message.content}`);
  console.log(
    `  Tokens: ${response.inputTokensConsumed} in, ${response.outputTokensConsumed} out`,
  );
  console.log();

  // Streaming chat
  console.log("🌊 Chat (streaming):");
  process.stdout.write("  ");
  for await (const token of client.chatStream({
    messages: [{ role: "user", content: "Write a haiku about JavaScript" }],
  })) {
    process.stdout.write(token);
  }
  console.log("\n");
}

main().catch(console.error);
```

## Run

```bash
WARP_API_KEY=sk-wp-your-key npx tsx index.ts
```
